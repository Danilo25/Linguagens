Int main() {

estrutura Usuario {
    String nome;
    Int idade;
    Flut salario;
}

Usuario u <- {
    nome = "pedro";
    idade = 32;
    salario = 3000.0;
}

Int x <- 1000, y <- 150;

for (x + y + u.salario < 5000) {
    x <- x + y;
    y <- y * 2;
}

u.salario <- x + y + u.salario;

pintef(praCadeia(u.salario));

retorne 0;

}
